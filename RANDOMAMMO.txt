ACTOR CartridgePinata : CartridgeBonus
{
	var int user_timer;
	var int user_curtime;
	-COUNTITEM
	+INVENTORY.ALWAYSPICKUP
	scale 0.4
	Gravity 0.65
	RenderStyle Add
	Inventory.PickupMessage "+5 Low Caliber Rounds"
	Inventory.PickupSound "misc/clip_PickUp"
	States
	{
	Spawn:
		TNT1 A 0 A_SetUserVar("user_curtime",0)
		TNT1 A 0 A_SetUserVar("user_timer",35*11)
	SpawnLoop:
		"####" "#" 1
		{
			if (user_timer >= DelayVacuum)
			{
				A_RearrangePointers(DefPtr,DefPtr,NullPtr,4);
				if (A_CheckProximity("Null","PlayerPawn",RadiusPull,1,CPXF_CHECKSIGHT|CPXF_SETTRACER|CPXF_ANCESTOR))
				{
					return state("Suckage");
				}
			}
			return state(0);
		}
		Loop
	Fade:
		PINA N 4 bright A_FadeOut(0.2)
		PINA N 4 A_FadeOut(0.2)
		loop
	Suckage:
		PINA N 4 bright
		{
			if(user_curtime > user_timer)
			{
				return state("Fade");
			}
			else
			{
				user_curtime += 4;
				return state(0);
			}
		}
		PINA N 4
		{
			if(user_curtime > user_timer)
			{
				return state("Fade");
			}
			else
			{
				user_curtime += 4;
				return state(0);
			}
		}
		"####" "#" 1
		{
			if (GetDistance(true,TracerPtr) <= sv_ammorange + 32 && CountInv("ExecutionToken",TracerPtr) == 0 && CountInv("DoGloryMelee",TracerPtr) == 0)
			{
				A_ChangeFlag(NOCLIP,true);
				A_FaceTracer(0,0,0,0,FAF_MIDDLE);
			}
			if (GetDistance(true,TracerPtr) <= sv_ammorange && CountInv("ExecutionToken",TracerPtr) == 0 && CountInv("DoGloryMelee",TracerPtr) == 0)
			{
				A_Warp(TracerPtr,0,0,16,0,WARPF_NOCHECKPOSITION|WARPF_INTERPOLATE|WARPF_STOP);
				A_ChangeFlag(NOCLIP,false);
				A_ChangeFlag(NOINTERACTION,false);
				A_ChangeFlag(NOBLOCKMAP,false);
			}
			return state(0);
		}
	Loop	
	}
}
